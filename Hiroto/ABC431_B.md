<!--
ロボットがあり、はじめロボットの重さは X です。 このロボットには、同時に取り付けられる部品が種類 1, 種類 2,…, 種類 N の N 種類あります。 種類 i (1≤i≤N) の部品の重さは W iです。 
はじめ、ロボットには N 種類のうちのどの部品もついていません。

次の Q 個のクエリを順に処理してください。 
i 番目 (1≤i≤Q) のクエリは、整数 P iで表される次のようなクエリです。

現在、ロボットに種類 P iの部品がついていない場合は取り付け、ついている場合は取り外す。その後、現在のロボットの重さを出力する。
-->

```c++
#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;

int main()
{
    int x, n;
    cin >> x >> n; // 初期重さと部品の種類

    vector<int> w(n, 0);       // 部品の重さ
    vector<int> p_check(n, 0); // 部品がついてるか

    for (int i = 0; i < n; i++)
    {
        cin >> w[i];
    }

    int q;
    cin >> q; // クエリの数
    for (int i = 0; i < q; i++)
    {
        int p;
        cin >> p; // クエリにて追加する部品
        p--;      // 0始まりに変換

        if (p_check[p] == 0)
        {                   // 部品がついてないなら
            x += w[p];      // 部品の重さを追加
            p_check[p] = 1; // 部品を追加(これ忘れてた)
        }
        else
        {                   // 部品がついてるなら
            x -= w[p];      // 部品の重さを取り外す
            p_check[p] = 0; // 部品を取り外す
        }
        cout << x << "\n";
    }
}
```
